== animal == 
    cd animal
    sudo docker stop animal; sudo docker rm animal ; sudo docker rmi animal
    rm  utils ; cp -r ../utils/ .; sudo docker build -t animal . ; sudo docker run --net=host -d --name animal animal; rm -rf utils ; ln -s ../utils utils
    cd ..

== utils ==
go to docker bash
    sudo docker exec -it <container> bash
check docker log
    sudo docker logs -f <container>

build zip <file> include <folder>
    zip -r <file> <folder> 

dump file system: 
    sudo docker export -o <File>.tar <container>
